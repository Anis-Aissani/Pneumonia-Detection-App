\section{Introduction et contexte}

\begin{frame}{Contexte et motivation}
\begin{itemize}
  \item La pneumonie reste une cause majeure de mortalite infectieuse.
  \item En radiologie, les examens sont souvent traites en FIFO.
  \item Risque: un cas grave peut attendre trop longtemps.
  \item Objectif: proposer un outil de triage automatise, explicable et tracable.
\end{itemize}
\vspace{4pt}
\textbf{Positionnement:} prototype academique d'aide a la priorisation, pas un dispositif medical certifie.
\end{frame}

\begin{frame}{Problematique et objectifs}
\textbf{Problematique:}
\begin{itemize}
  \item Integrer un modele ML pre-entraine dans une architecture web securisee.
  \item Garantir la tracabilite des analyses et la gestion multi-roles.
  \item Rejeter les entrees hors distribution (photos, documents non RX).
\end{itemize}
\vspace{4pt}
\textbf{Objectifs techniques:}
\begin{itemize}
  \item API FastAPI robuste + interface Vue.js 3.
  \item Triage automatique \texttt{CRITICAL / MODERATE / ROUTINE}.
  \item Dashboard BI + generation de rapport PDF.
\end{itemize}
\end{frame}

\begin{frame}{Architecture backend en couches}
\begin{itemize}
  \item \textbf{Controller layer:} routes FastAPI dans \texttt{main.py}.
  \item \textbf{Service layer:} orchestration metier dans \texttt{services.py}.
  \item \textbf{Data layer:} persistance et triage dans \texttt{database.py}.
  \item \textbf{ML layer:} OOD, inference et heatmap dans \texttt{predictor.py}.
  \item \textbf{Config layer:} environnement centralise dans \texttt{settings.py}.
\end{itemize}
\vspace{4pt}
\textbf{Resultat:} separation claire des responsabilites, code plus lisible et testable.
\end{frame}

\begin{frame}{Docker Compose: port 3000 explicite}
\begin{itemize}
  \item Commande unique de demo: \texttt{docker compose up --build}.
  \item Interface frontend exposee sur \textbf{http://localhost:3000}.
  \item API backend exposee sur \texttt{http://localhost:8000}.
  \item Les logs frontend affichent le lien cliquable \texttt{http://localhost:3000} au demarrage.
\end{itemize}
\vspace{3pt}
\textbf{Point jury:} Docker est le levier d'industrialisation principal du prototype.
\end{frame}

\section{Implementation et resultats}

\begin{frame}{Choix algorithmiques et triage clinique}
\begin{itemize}
  \item Modele retenu: \texttt{EfficientNet-B0 + SVM} (meilleur compromis robustesse/performance).
  \item Score test binome: \textbf{Accuracy 86.2\%} et \textbf{F1 88.4\%}.
  \item Fallback technique deploye: \texttt{HOG + PCA + SVM}.
\end{itemize}
\[
\text{CRITICAL} = (\hat{y}=\text{PNEUMONIA}) \land (p > 0.85)
\]
\[
\text{MODERATE} = (\hat{y}=\text{PNEUMONIA}) \land (0.15 \le p \le 0.85)
\]
\[
\text{ROUTINE} = (\hat{y}=\text{NORMAL}) \lor (\hat{y}=\text{PNEUMONIA} \land p < 0.15)
\]
\end{frame}
